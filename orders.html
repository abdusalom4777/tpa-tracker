В doGet добавьте:

if (action === "getOrders") {
  var sheet = ss.getSheetByName("Заказы");
  var orders = [];
  if (sheet && sheet.getLastRow() > 1) {
    var data = sheet.getRange(2,1,sheet.getLastRow()-1,14).getValues();
    data.forEach(function(row){
      if(row[0]) orders.push({
        id:String(row[0]), client:String(row[1]||""),
        moldId:String(row[2]||""), partName:String(row[3]||""),
        qty:Number(row[4]||0), prodHours:Number(row[5]||0),
        machine:String(row[6]||""), priority:String(row[7]||"normal"),
        deadline:String(row[8]||""), eta:String(row[9]||""),
        status:String(row[10]||"new"), comment:String(row[11]||""),
        createdAt:String(row[12]||""), updatedAt:String(row[13]||"")
      });
    });
  }
  var json = JSON.stringify({status:"ok",orders:orders});
  if(callback) return ContentService.createTextOutput(callback+"("+json+")").setMimeType(ContentService.MimeType.JAVASCRIPT);
  return ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON);
}
