function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var d = JSON.parse(e.postData.contents); // Добавить пресс-форму if (d.action === "addMold") { var sheet = ss.getSheetByName("Пресс-формы") || ss.insertSheet("Пресс-формы"); if (sheet.getLastRow() === 0) sheet.appendRow(["№ Пресс-формы","Название","Станок"]); if (sheet.getLastRow() > 1) { var ids = sheet.getRange(2,1,sheet.getLastRow()-1,1).getValues().map(function(r){return String(r[0]);}); if (ids.indexOf(d.id) >= 0) return ContentService.createTextOutput(JSON.stringify({status:"error",msg:"Уже существует"})).setMimeType(ContentService.MimeType.JSON); } sheet.appendRow([d.id, d.label||"", d.machine||""]); return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON); } // Удалить пресс-форму if (d.action === "deleteMold") { var sheet = ss.getSheetByName("Пресс-формы"); if (sheet && sheet.getLastRow() > 1) { var data = sheet.getRange(2,1,sheet.getLastRow()-1,1).getValues(); for (var i = data.length-1; i >= 0; i--) { if (String(data[i][0]) === String(d.id)) { sheet.deleteRow(i+2); break; } } } return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON); } // Добавить запись в журнал (по умолчанию) var sheet = ss.getSheetByName("Журнал") || ss.insertSheet("Журнал"); if (sheet.getLastRow() === 0) { sheet.appendRow(["Дата","Смена","Оператор","ТПА","Материал","Деталь","Пресс-форма","Гнёзда","Циклов","Время цикла (с)","Годные","Брак","% брака","Вес дет. 4
Сохраните и опубликуйте скрипт
1. Нажмите 💾 Сохранить проект (Ctrl+S)
2. Нажмите Развернуть → Новое развёртывание
3. Тип: Веб-приложение
4. «Выполнять как»: Я (ваш аккаунт)
5. «Доступ»: Все (Anyone)
6. Нажмите Развернуть → разрешите доступ
7. Скопируйте URL веб-приложения
5
Создайте лист «Пресс-формы» в таблице
Внизу таблицы нажмите «+» (добавить лист), назовите его точно: Пресс-формы
Заполните данные:
— Столбец A: № пресс-формы (напр. ПФ-01, ПФ-02...)
— Столбец B: Название / описание (необязательно)

Приложение автоматически загрузит этот список при открытии. Кнопка 🔄 рядом с полем — обновить вручную.
6
Вставьте URL в поле выше
Вернитесь на вкладку «Google», вставьте скопированный URL в поле «URL скрипта» — и всё готово! Теперь каждая новая запись будет автоматически отправляться в таблицу, а список пресс-форм загружаться при открытии.
